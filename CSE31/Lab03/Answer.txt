1) #include <stdio.h>  // static memory
   #include <stdlib.h> // static
   #include <malloc.h> // static

   int main() {      // stack
       int num;	     // stack
       int *ptr;     // stack
       int **handle; // stack

       num = 14;     // stack
       ptr = (int *)malloc(2 * sizeof(int)); // heap
       handle = &ptr;  // stack
       *ptr = num;     // stack
       ptr = &num;     // stack
       handle = (int **)malloc(1 * sizeof(int **));  // heap
}
2) In this program, memory doesn't get freed, but it gets allocated by malloc
3) This memory is in somewhere of heap
4) It is possible because this code never freed
